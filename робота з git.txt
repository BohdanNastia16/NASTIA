git init - создаєм локальний репозиторій
git add ./ - сохраняєм всі файли для того щоб потім ми могли їх зкомітить
git status - перевіряєм в якому файлі відбулись зміни
git commit -m "пишем якийсь текст" - сохраняєм всі файли які підготовили
git push -u  origin master - відправили всі наші файли на онлайн репозиторій
git pull origin - скачуєм собі файли
